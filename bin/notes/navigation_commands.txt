cd - изменить каталог cd test
dir - вывести список каталогов  dir c:\test\directory
copy - копировать файл	copy c:\test\test.txt c:\windows\test.txt
move - переместить файл	move c:\test\test.txt c:\windows\test.txt
mkdir (или md) - создать новый каталог	mkdir testdirectory
rmdir (или rd) - удалить директорию	rmdir c:\test\directory
del - удалить файл del c:\test\test.txt
cls - очистить содержимое терминала
[CMD] /? - справка для команды	cd /?
pwd - показать путь рабочей директории
touch file_name - открыть файл, или создать его, если он не существует
export var_name = "Katya" - установить переменную окружения
'var_name, hello' - считывает как текст
"var_name, hello" - осуществит интерполяцию переменной

Ctrl+c - прекратить работу

Список всех команд: https://ss64.com/
